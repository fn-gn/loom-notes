\section{Conditioning: the tax, named}\label{sec:cond}
\warmth{0}\quad\whisper{Curve the bowl up everywhere and descent goes geometric. The rate is set by $\kappa=L/\mu$.}

\begin{strand}
Add one more assumption, \keyword{strong convexity}: the bowl curves up at rate at least $\mu$
everywhere, so it has a unique minimum and walls that push back. Now gradient descent converges
\keyword{linearly} (the error shrinks by a constant factor each step), and the factor is governed
by a single number, the \keyword{condition number} $\kappa=L/\mu$. This is the tax.
\end{strand}

\block{Strong convexity and the linear rate}

\begin{definition}[$\mu$-strong convexity]\label{def:strong}
$f$ is \keyword{$\mu$-strongly convex} if it lies above each tangent parabola:
$f(y)\ge f(x)+\inner{\nabla f(x)}{y-x}+\tfrac{\mu}{2}\norm{y-x}^2$. Sandwiched with $L$-smoothness,
the curvature is pinned into $[\mu,L]$.
\end{definition}

\begin{theorem}[linear convergence]\label{thm:linear}
If $f$ is $\mu$-strongly convex and $L$-smooth, gradient descent with $\eta=1/L$ obeys
\[
  \norm{x_k-\xstar}^2\;\le\;\Big(1-\frac{\mu}{L}\Big)^{\!k}\,\norm{x_0-\xstar}^2
  \;=\;\Big(1-\frac1\kappa\Big)^{\!k}\,\norm{x_0-\xstar}^2 .
\]
\end{theorem}
\begin{proof}
Skeleton: show one step is a contraction, $\norm{x_{k+1}-\xstar}\le(1-\mu/L)^{1/2}\norm{x_k-\xstar}$,
using strong convexity and smoothness to bound $\inner{\nabla f(x_k)}{x_k-\xstar}$ from below.

\end{proof}

\begin{strand}
Picture the level sets. With curvatures in $[\mu,L]$ the bowl is an ellipse whose axis ratio is
$\kappa=L/\mu$. When $\kappa=1$ it is a perfect circle and descent points straight at the minimum,
done in one step. When $\kappa$ is large it is a long thin valley: the gradient points mostly
\emph{across} the valley, not along it, so the iterates \keyword{zigzag} and crawl toward the bottom.
Ill-conditioning, not dimension, is what makes gradient descent slow.
\end{strand}

\begin{yourturn}
Turn the rate into a step count. Since $(1-1/\kappa)^k\le e^{-k/\kappa}$,
\[
  k\;=\;O\!\big(\kappa\,\log\tfrac1\varepsilon\big)\ \text{steps for error }\varepsilon.
\]
\begin{itemize}[leftmargin=1.3em,itemsep=1pt]
  \item A round bowl $\kappa=1$ needs about \fillin[2cm] steps; a valley with $\kappa=10^4$ needs about \fillin[1.6cm].
  \item For a quadratic, $\kappa$ is the ratio $\lambda_{\max}/\lambda_{\min}$ of the Hessian, i.e.\ its \fillin[3cm].
\end{itemize}
\recall{$O(\log\tfrac1\varepsilon)$ vs.\ $O(10^4\log\tfrac1\varepsilon)$; and $\kappa$ is the Hessian's \emph{condition number} (the SVD's $\sigma_{\max}/\sigma_{\min}$). The valley costs you a factor of $\kappa$.}
\end{yourturn}

\begin{strand}
Compare §\ref{sec:smooth}: convex-only gives $O(1/\varepsilon)$, strongly convex gives
$O(\kappa\log\tfrac1\varepsilon)$. The log is a huge win (linear vs.\ sublinear); the $\kappa$ in
front is the price still to be haggled down in §\ref{sec:faces}.
\end{strand}
