\section{Paying the tax down}\label{sec:faces}
\warmth{0}\quad\whisper{Momentum buys $\sqrt\kappa$; stochasticity buys cheap steps; Newton erases $\kappa$ at a price.}

\begin{strand}
The condition number $\kappa$ is the bill. The art of optimization is haggling it down. Three
moves dominate: \keyword{momentum} (remember where you were going, and reduce $\kappa$ to
$\sqrt\kappa$), \keyword{stochasticity} (use a cheap noisy gradient on huge data), and
\keyword{preconditioning / Newton} (reshape the bowl to a circle, erasing $\kappa$ entirely).
\end{strand}

\block{The landscape of rates}
\noindent\begin{tabularx}{\linewidth}{@{}l l L@{}}
\toprule
{\color{indigo}method / setting} & {\color{indigo}steps to $\varepsilon$} & \multicolumn{1}{l}{\color{indigo}what it buys}\\
\midrule
GD, convex             & $O(1/\varepsilon)$           & baseline, sublinear\\
GD, strongly convex    & $O(\kappa\log\tfrac1\varepsilon)$ & linear, but taxed by $\kappa$\\
Nesterov accelerated   & $O(\sqrt\kappa\log\tfrac1\varepsilon)$ & \keyword{optimal} first-order rate\\
SGD (strongly convex)  & $O(1/\varepsilon)$          & each step $O(1)$ not $O(n)$; noise floor needs decreasing $\eta$\\
GD, nonconvex smooth   & $O(1/\varepsilon^2)$        & only a \emph{stationary} point $\norm{\nabla f}\le\varepsilon$\\
\bottomrule
\end{tabularx}

\begin{theorem}[acceleration]\label{thm:nesterov}
For $\mu$-strongly convex, $L$-smooth $f$, Nesterov's momentum method converges as
$(1-1/\sqrt\kappa)^k$, needing only $O(\sqrt\kappa\,\log\tfrac1\varepsilon)$ steps. No first-order
method (one gradient per step) can beat $\sqrt\kappa$ on the worst such $f$.
\end{theorem}

\recall{Momentum's trick: extrapolate a little past the current point before taking the gradient, so consistent progress builds speed and the cross-valley zigzag cancels. It is the matching upper bound to a genuine $\Omega(\sqrt\kappa)$ lower bound.}

\begin{yourturn}
Count the payoff on a stiff problem, $\kappa=10^4$ (ignore the $\log$ factor).
\[
  \text{plain GD: } \approx\fillin[1.5cm]\text{ steps},\qquad
  \text{accelerated: } \approx\fillin[1.5cm]\text{ steps},\qquad
  \text{speedup } \approx\fillin[1.2cm].
\]
\recall{$\kappa=10^4$ vs.\ $\sqrt\kappa=10^2$, a $100\times$ speedup. On ill-conditioned problems momentum is not a tweak, it is the difference between feasible and not.}
\end{yourturn}

\block{The one idea, recovered}
\begin{strand}
Read the master table again. Every rate is the same story told at a different conditioning. A round
bowl ($\kappa\!=\!1$) is trivial; the tax grows with $\kappa$; momentum softens it to $\sqrt\kappa$;
a good \keyword{preconditioner} (in the limit, Newton's method, which rescales by the inverse
Hessian) reshapes the bowl back to a circle and removes $\kappa$ altogether, at the cost of solving
a linear system each step. \keyword{The rate is a tax on conditioning, and optimization is the art
of paying less.}
\end{strand}

\loose{The machine-learning reality: $f$ is nonconvex and $n$ is enormous, so practice is SGD with momentum and \keyword{adaptive} per-coordinate steps (Adam) that cheaply approximate preconditioning. The same three levers, conditioning, momentum, stochasticity, are still exactly what is being traded.}
